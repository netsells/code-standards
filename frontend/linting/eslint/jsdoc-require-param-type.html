# jsdoc/require-param-type warn

Requires that each @param tag has a type value.

# Examples

❌ Incorrect

/**
 * @param foo
 */
function quux (foo) {

}
// Message: Missing JSDoc @param "foo" type.

✅ Correct

/**
 *
 */
function quux1 (foo) {

}

/**
 * @param {number} foo
 */
function quux2 (foo) {

}
